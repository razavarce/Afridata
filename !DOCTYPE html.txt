    {id: 1, name: 'Administrador', login: 'admin', password: 'admin', role: 'admin'},
    {id: 2, name: 'Usuario Lectura', login: 'viewer', password: 'viewer', role: 'viewer'}
];
let nextUserId = 3;

function createParticles() {
    const container = document.getElementById('particles');
    for(let i = 0; i < 30; i++) {
        const p = document.createElement('div');
        p.className = 'particle';
        p.style.left = Math.random() * 100 + '%';
        p.style.animationDuration = (8 + Math.random() * 15) + 's';
        p.style.animationDelay = Math.random() * 10 + 's';
        p.style.width = (2 + Math.random() * 4) + 'px';
        p.style.height = p.style.width;
        container.appendChild(p);
    }
}

function doLogin() {
    const u = document.getElementById('loginUser').value;
    const p = document.getElementById('loginPass').value;
    const user = users.find(user => user.login === u && user.password === p);
    if(user) {
        currentUser = user;
        document.getElementById('loginPage').style.display = 'none';
        document.getElementById('appPage').style.display = 'block';
        
        document.getElementById('headerUsername').textContent = user.name;
        document.getElementById('headerRole').textContent = user.role === 'admin' ? 'Admin' : 'Lector';
        document.getElementById('headerRole').className = 'role-badge ' + (user.role === 'admin' ? 'role-admin' : 'role-viewer');
        
        setTimeout(() => {
            initMap();
            loadPaisesTable();
            loadAutoridades();
            loadEmbajadas();
            populateCountrySelect();
            loadUsers();
            updateAdminAccess();
            renderBilateralContent();
            renderMultilateralContent();
        }, 300);
    } else {
        document.getElementById('loginError').style.display = 'block';
    }
}

function doLogout() {
    currentUser = null;
    document.getElementById('loginPage').style.display = 'flex';
    document.getElementById('appPage').style.display = 'none';
    if(map) { map.remove(); map = null; }
}

function updateAdminAccess() {
    const isAdmin = currentUser && currentUser.role === 'admin';
    document.getElementById('adminContent').style.display = isAdmin ? 'block' : 'none';
    document.getElementById('viewerMessage').style.display = isAdmin ? 'none' : 'block';
    document.getElementById('editModeIndicator').classList.toggle('active', isAdmin);
    
    if(isAdmin) {
        document.getElementById('adminEditSection').style.display = 'block';
    }
}

function initMap() {
    map = L.map('map', {
        zoomControl: false,
        minZoom: 2,
        maxZoom: 8
    }).setView([5, 20], 3);
    
    L.control.zoom({ position: 'topright' }).addTo(map);
    
    L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
        attribution: '© OpenStreetMap | © CARTO | AFRIDATA MPPRE',
        maxZoom: 18
    }).addTo(map);
    
    setTimeout(() => map.invalidateSize(), 400);
    addMarkers('all');
}

function addMarkers(region) {
    if(markers.length) {
        markers.forEach(m => { try { map.removeLayer(m); } catch(e) {} });
        markers = [];
    }
    
    const filtered = region === 'all' ? paisesDB : paisesDB.filter(p => p.region === region);
    
    filtered.forEach(p => {
        const color = regionColors[p.region];
        const flagUrl = `https://flagcdn.com/w80/${p.bandera}.png`;
        
        const icon = L.divIcon({
            html: `<div class="custom-marker" style="width:${18}px;height:${18}px;">
                <div class="marker-dot" style="background:${color};"></div>
            </div>`,
            className: '', iconSize: [18,18], iconAnchor: [9,9]
        });
        
        const marker = L.marker([p.lat, p.lng], {icon}).addTo(map);
        
        const hoverContent = `
            <div class="tooltip-content">
                <img class="tooltip-flag" src="${flagUrl}" alt="${p.nombre}" onerror="this.style.display='none'">
                <div>
                    <div class="tooltip-name">${p.nombre}</div>
                    <div class="tooltip-region">${regionNames[p.region]}</div>
                </div>
            </div>
        `;
        
        const tooltip = L.tooltip({
            permanent: false,
            direction: 'top',
            offset: [0, -10],
            className: 'country-hover-tooltip',
            sticky: true
        }).setContent(hoverContent);
        
        marker.bindTooltip(tooltip);
        
        marker.bindPopup(`
            <div class="popup-content">
                <img class="popup-flag-img" src="${flagUrl}" alt="${p.nombre}" onerror="this.style.display='none'">
                <div class="popup-name">${p.nombre}</div>
                <div class="popup-capital"><i class="fas fa-map-pin"></i> ${p.capital}</div>
                <div class="popup-president"><i class="fas fa-user"></i> ${p.presidente}</div>
                <div style="font-size:11px;color:${color};font-weight:700;margin-top:4px;">${regionNames[p.region]}</div>
                <button class="popup-btn" onclick="openCountryModal(${p.id})"><i class="fas fa-info-circle"></i> Ver Ficha Completa</button>
            </div>
        `, { maxWidth: 260, closeButton: true });
        
        marker.on('click', function() {
            this.openPopup();
        });
        
        markers.push(marker);
    });
    
    if(filtered.length > 0 && region !== 'all') {
        const group = L.featureGroup(markers);
        map.fitBounds(group.getBounds().pad(0.4));
    } else if(region === 'all') {
        map.setView([5, 20], 3);
    }
}

function filterRegion(region, btn) {
    currentRegion = region;
    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    addMarkers(region);
}

function filterCountries() {
    const q = document.getElementById('searchBox').value.toLowerCase().trim();
    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
    document.querySelector('.filter-btn').classList.add('active');

    if(markers.length) markers.forEach(m => { try { map.removeLayer(m); } catch(e) {} });
    markers = [];

    const filtered = paisesDB.filter(p =>
        p.nombre.toLowerCase().includes(q) ||
        p.capital.toLowerCase().includes(q) ||
        p.presidente.toLowerCase().includes(q) ||
        p.region.toLowerCase().includes(q)
    );

    filtered.forEach(p => {

            </div>`,
            className: '', iconSize: [20,20], iconAnchor: [10,10]
        });
        
        const marker = L.marker([p.lat, p.lng], {icon}).addTo(map);
        
        const hoverContent = `
            <div class="tooltip-content">
                <img class="tooltip-flag" src="${flagUrl}" alt="${p.nombre}" onerror="this.style.display='none'">
                <div>
                    <div class="tooltip-name">${p.nombre}</div>
                    <div class="tooltip-region">${regionNames[p.region]}</div>
                </div>
            </div>
        `;
        
        const tooltip = L.tooltip({
            permanent: false,
            direction: 'top',
            offset: [0, -10],
            className: 'country-hover-tooltip',
            sticky: true
        }).setContent(hoverContent);
        
        marker.bindTooltip(tooltip);
        
        marker.bindPopup(`
            <div class="popup-content">
                <img class="popup-flag-img" src="${flagUrl}" alt="${p.nombre}" onerror="this.style.display='none'">
                <div class="popup-name">${p.nombre}</div>

    if(filtered.length > 0) {
        const group = L.featureGroup(markers);
        map.fitBounds(group.getBounds().pad(0.4));
    }
}

function showModule(mod) {
    document.querySelectorAll('.module-btn').forEach(b => b.classList.remove('active'));
    event.target.closest('.module-btn').classList.add('active');

    document.querySelectorAll('.module-panel').forEach(p => p.classList.remove('active'));
    const panel = document.getElementById('panel-'+mod);
    if(panel) panel.classList.add('active');

    if(mod === 'mapa') {
        document.getElementById('modulesContainer').style.display = 'none';
        document.getElementById('appLayout').style.display = 'flex';
        setTimeout(() => { if(map) map.invalidateSize(); }, 200);
    } else {
        document.getElementById('modulesContainer').style.display = 'block';
        document.getElementById('appLayout').style.display = 'none';
    }
    
    if(mod === 'admin') {
        updateAdminAccess();
    }
    
    if(mod === 'bilateral') {
        renderBilateralContent();
    }
    
    if(mod === 'multilateral') {
        renderMultilateralContent();
    }
    
    if(mod === 'reportes') {
        document.getElementById('reportDate').textContent = new Date().toLocaleDateString('es-VE', {
            weekday: 'long', year: 'numeric', month: 'long', day: 'numeric'
        });
    }
}

function renderBilateralContent() {
    const container = document.getElementById('bilateralContent');
    if(!container) return;
    
    const search = document.getElementById('bilateralSearch')?.value.toLowerCase() || '';
    const regionFilter = document.querySelector('#regionSelector .region-tab.active')?.textContent.trim().toLowerCase() || 'todas las regiones';
    const statusFilter = document.querySelector('#statusFilters .filter-chip.active')?.textContent.trim() || 'todos';
    
    let filteredCountries = paisesDB.filter(p => {
        const matchesSearch = p.nombre.toLowerCase().includes(search) || 
                             p.capital.toLowerCase().includes(search) ||
                             regionNames[p.region].toLowerCase().includes(search);
        
        let matchesRegion = true;
        if(regionFilter.includes('norte')) matchesRegion = p.region === 'norte';
        else if(regionFilter.includes('occidental')) matchesRegion = p.region === 'occidental';
        else if(regionFilter.includes('central')) matchesRegion = p.region === 'central';
        else if(regionFilter.includes('oriental')) matchesRegion = p.region === 'oriental';
        else if(regionFilter.includes('austral')) matchesRegion = p.region === 'austral';
        
        let matchesStatus = true;
        const bd = bilateralData[p.id];
        if(statusFilter.includes('con relaciones') || statusFilter.includes('🟢')) matchesStatus = bd?.status === 'active';
        else if(statusFilter.includes('limitadas') || statusFilter.includes('🟡')) matchesStatus = bd?.status === 'limited';
        else if(statusFilter.includes('sin relación') || statusFilter.includes('🔴')) matchesStatus = bd?.status === 'none';
        else if(statusFilter.includes('embajada') || statusFilter.includes('🏛️')) matchesStatus = bd?.hasEmbassy === true;
        
        return matchesSearch && matchesRegion && matchesStatus;
    });
    
    // Agrupar por región
    const grouped = {};
    filteredCountries.forEach(p => {
        if(!grouped[p.region]) grouped[p.region] = [];
        grouped[p.region].push(p);
    });
    
    let html = '';
    
    const regionOrder = ['norte', 'occidental', 'central', 'oriental', 'austral'];
    
    regionOrder.forEach(region => {
        if(!grouped[region]) return;
        
        html += `<div class="report-section">
            <h3 style="color:${regionColors[region]}"><i class="fas fa-globe-africa"></i> ${regionNames[region]} (${grouped[region].length} países)</h3>
            <div class="bilateral-grid">`;
        
        grouped[region].forEach(p => {
            const bd = bilateralData[p.id] || { status: 'none', hasEmbassy: false, strategicLevel: 'baja', date: '', agreements: [], strategic: [], coopAreas: [] };
            const flagUrl = `https://flagcdn.com/w48/${p.bandera}.png`;
            const statusClass = bd.status === 'active' ? 'status-active' : bd.status === 'limited' ? 'status-limited' : 'status-none';
            const statusText = bd.status === 'active' ? '🟢 Relaciones Activas' : bd.status === 'limited' ? '🟡 Limitadas' : '🔴 Sin Relación';
            const strategicText = bd.strategicLevel === 'alta' ? '⭐ Alta' : bd.strategicLevel === 'media' ? '📊 Media' : '📉 Baja';
            
            html += `
                <div class="country-bilateral-card" data-country-id="${p.id}">
                    <div class="country-bilateral-header" onclick="toggleBilateralCard(${p.id})">
                        <img src="${flagUrl}" alt="${p.nombre}" onerror="this.style.display='none'">
                        <div class="country-info">
                            <div class="country-name">${p.nombre}</div>
                            <div class="country-region">${p.capital} • ${strategicText}</div>
                        </div>
                        <span class="status-badge ${statusClass}">${statusText}</span>
                        <div class="toggle-icon"><i class="fas fa-chevron-down"></i></div>
                    </div>
                    <div class="country-bilateral-content" id="bilateral-content-${p.id}">
                        <div class="bilateral-section">
                            <div class="bilateral-section-title"><i class="fas fa-handshake"></i> Información General</div>
                            <div class="diplomatic-info">
                                <div class="dipl-row"><span class="dipl-label">Relaciones desde:</span><span class="dipl-value">${bd.date || 'N/A'}</span></div>
                                <div class="dipl-row"><span class="dipl-label">Embajada:</span><span class="dipl-value">${bd.hasEmbassy ? '🏛️ Sí - ' + p.embajada : '❌ No'}</span></div>
                                <div class="dipl-row"><span class="dipl-label">Nivel Estratégico:</span><span class="dipl-value">${strategicText}</span></div>
                                <div class="dipl-row"><span class="dipl-label">Acuerdos Vigentes:</span><span class="dipl-value">${bd.agreements?.length || 0}</span></div>
                            </div>
                        </div>
                        
                        ${bd.agreements && bd.agreements.length > 0 ? `
                        <div class="bilateral-section">
                            <div class="bilateral-section-title"><i class="fas fa-file-contract"></i> Acuerdos Bilaterales (${bd.agreements.length})</div>
                            ${bd.agreements.map((ag, i) => `
                                <div class="agreement-item">
                                    <div class="agreement-title">${ag.title}</div>
                                    <div class="agreement-desc">${ag.desc}</div>
                                    <div class="agreement-meta">
                                        <span><i class="fas fa-calendar"></i> ${ag.date}</span>
                                        <span class="badge ${ag.status === 'Vigente' ? 'badge-green' : 'badge-orange'}">${ag.status}</span>
                                    </div>
                                </div>
                            `).join('')}
                        </div>
                        ` : '<div class="bilateral-section"><div class="bilateral-section-title"><i class="fas fa-file-contract"></i> Acuerdos</div><p style="color:var(--text-muted);font-size:13px;">No se registran acuerdos bilaterales vigentes.</p></div>'}
                        
                        <div class="bilateral-section">
                            <div class="bilateral-section-title"><i class="fas fa-star"></i> Relación Estratégica</div>
                            <div style="margin-bottom:10px;">

    p.embajada = document.getElementById('editEmbassy').value;

    p.economia = document.getElementById('editEconomy').value;
    
    loadPaisesTable();

    showToast('País actualizado exitosamente', 'success');
}

function createUser() {
    const name = document.getElementById('newUserName').value;
    const login = document.getElementById('newUserLogin').value;
    const pass = document.getElementById('newUserPass').value;
    const role = document.getElementById('newUserRole').value;
    
    if(!name || !login || !pass) {
        showToast('Completa todos los campos', 'error');
        return;
    }
    
    if(users.find(u => u.login === login)) {
        showToast('El nombre de usuario ya existe', 'error');
        return;
    }
    
    users.push({id: nextUserId++, name, login, password: pass, role});
    loadUsers();
    
    document.getElementById('newUserName').value = '';
    document.getElementById('newUserLogin').value = '';
    document.getElementById('newUserPass').value = '';
    
    showToast('Usuario creado exitosamente', 'success');
}

function loadUsers() {
    const container = document.getElementById('userList');
    container.innerHTML = users.map(u => `
        <div class="user-item">
            <div class="user-info">
                <div class="user-avatar ${u.role}">
                    ${u.role === 'admin' ? '⚙️' : '👁️'}
                </div>
                <div class="user-details">
                    <h4>${u.name}</h4>
                    <p>@${u.login} • <span class="role-badge ${u.role === 'admin' ? 'role-admin' : 'role-viewer'}">${u.role === 'admin' ? 'Administrador' : 'Solo Lectura'}</span></p>
                </div>
            </div>
            <div class="user-actions">
                ${u.login !== 'admin' ? `<button class="action-btn delete" onclick="deleteUser(${u.id})" title="Eliminar"><i class="fas fa-trash"></i></button>` : ''}
            </div>
        </div>
    `).join('');
}

function deleteUser(id) {
    if(id === 1) {
        showToast('No puedes eliminar el administrador principal', 'error');
        return;
    }
    users = users.filter(u => u.id !== id);
    loadUsers();
    showToast('Usuario eliminado', 'success');
}

function showToast(message, type = 'info') {
    const existing = document.querySelector('.toast');
    if(existing) existing.remove();
    
    const toast = document.createElement('div');
    toast.className = `toast ${type}`;
    toast.innerHTML = `
        <i class="fas fa-${type === 'success' ? 'check-circle' : type === 'error' ? 'exclamation-circle' : 'info-circle'}"></i>
        <span>${message}</span>
    `;
    document.body.appendChild(toast);
    
    setTimeout(() => toast.remove(), 3000);
}

function generateReport(type) {
    const { jsPDF } = window.jspdf;
    const doc = new jsPDF();
    
    doc.setFontSize(20);
    doc.setTextColor(26, 82, 118);
    doc.text('AFRIDATA - Reporte MPPRE', 105, 20, {align: 'center'});
    
    doc.setFontSize(12);
    doc.setTextColor(100);
    doc.text(`Generado: ${new Date().toLocaleDateString('es-VE')}`, 105, 30, {align: 'center'});
    
    doc.setDrawColor(0, 212, 255);
    doc.line(20, 35, 190, 35);
    
    if(type === 'resumen') {
        doc.setFontSize(16);
        doc.setTextColor(26, 82, 118);
        doc.text('Resumen Ejecutivo por Región', 20, 50);
        
        doc.autoTable({
            startY: 55,
            head: [['Región', 'Países', 'Rel. Diplomáticas', 'Embajadas', 'Acuerdos']],
            body: [
                ['África del Norte', '7', '6', '2', '12'],
                ['África Occidental', '16', '14', '5', '18'],
                ['África Central', '9', '7', '3', '8'],
                ['África Oriental', '15', '10', '5', '15'],
                ['África Austral', '8', '6', '3', '10']
            ],
            theme: 'grid',
            headStyles: {fillColor: [26, 82, 118]},
            styles: {fontSize: 10}
        });
        
        doc.addPage();
        doc.setFontSize(16);
        doc.text('Top 5 Países por Población', 20, 20);
        
        doc.autoTable({
            startY: 25,
            head: [['País', 'Población (M)', 'Presidente', 'Gobierno']],
            body: [
                ['Nigeria', '230.9', 'Bola Ahmed Tinubu', 'República Federal'],
                ['Etiopía', '129.7', 'Abiy Ahmed', 'República Federal'],
                ['R.D. del Congo', '105.0', 'Félix Tshisekedi', 'República Semipresidencial'],
                ['Egipto', '114.5', 'Abdelfatah El-Sisi', 'República Semipresidencial'],
                ['Tanzania', '69.4', 'Samia Suluhu Hassan', 'República Semipresidencial']
            ],
            theme: 'grid',
            headStyles: {fillColor: [26, 82, 118]},
            styles: {fontSize: 10}
        });
    } else if(type === 'bilateral') {
        doc.setFontSize(16);
        doc.setTextColor(26, 82, 118);
        doc.text('Relaciones Bilaterales Venezuela-África', 20, 50);
        
        const bilateralTableData = [];
        paisesDB.forEach(p => {
            const bd = bilateralData[p.id];
            if(bd && bd.agreements && bd.agreements.length > 0) {
                bd.agreements.forEach(ag => {
                    bilateralTableData.push([p.nombre, ag.title, ag.status, ag.date]);
                });
            }
        });
        
        doc.autoTable({
            startY: 55,
            head: [['País', 'Acuerdo', 'Estado', 'Fecha']],
            body: bilateralTableData.slice(0, 50),
            theme: 'grid',
            headStyles: {fillColor: [26, 82, 118]},
            styles: {fontSize: 8},
            margin: {top: 55}
        });
    } else if(type === 'multilateral') {
        doc.setFontSize(16);
        doc.setTextColor(26, 82, 118);
        doc.text('Participación Multilateral Venezuela-África', 20, 50);
        
        let y = 55;
        Object.keys(multilateralData).forEach(key => {
            const org = multilateralData[key];
            if(y > 250) {
                doc.addPage();
                y = 20;
            }
            doc.setFontSize(14);
            doc.setTextColor(0, 212, 255);
            doc.text(`${org.icon} ${org.name}`, 20, y);
            y += 8;
            doc.setFontSize(10);
            doc.setTextColor(100);
            doc.text(org.description.substring(0, 100) + '...', 20, y);
            y += 15;
        });
    } else if(type === 'paises') {
        doc.setFontSize(16);
        doc.setTextColor(26, 82, 118);
        doc.text('Listado Completo de Países Africanos', 20, 50);
        
        const tableData = paisesDB.map(p => [p.nombre, p.capital, p.presidente, p.poblacion.toString(), regionNames[p.region]]);
        
        doc.autoTable({
            startY: 55,
            head: [['País', 'Capital', 'Presidente', 'Población (M)', 'Región']],
            body: tableData,
            theme: 'grid',
            headStyles: {fillColor: [26, 82, 118]},
            styles: {fontSize: 8},
            didDrawPage: function(data) {
                doc.setFontSize(10);
                doc.setTextColor(100);
                doc.text('AFRIDATA - MPPRE', 105, 290, {align: 'center'});
            }
        });
    }
    
    const fileName = `AFRIDATA_Reporte_${type}_${new Date().toISOString().split('T')[0]}.pdf`;
    doc.save(fileName);
    showToast(`Reporte PDF generado: ${fileName}`, 'success');
    closeDownloadModal();
}

function exportExcel() {
    // Crear CSV como alternativa simple
    let csv = 'País,Capital,Presidente,Gobierno,Población (M),Región,Idioma,Moneda,Embajada\n';
    paisesDB.forEach(p => {
        csv += `"${p.nombre}","${p.capital}","${p.presidente}","${p.gobierno}",${p.poblacion},"${regionNames[p.region]}","${p.idioma}","${p.moneda}","${p.embajada}"\n`;
    });
    
    const blob = new Blob(['\ufeff' + csv], { type: 'text/csv;charset=utf-8;' });
    const link = document.createElement('a');
    link.href = URL.createObjectURL(blob);
    link.download = `AFRIDATA_Paises_${new Date().toISOString().split('T')[0]}.csv`;
    link.click();
    showToast('Archivo CSV exportado exitosamente', 'success');
    closeDownloadModal();
}
